import typer, time, asyncio, json
from pathlib import Path
from orchestrator.graph import TestGraph
from orchestrator.control_room import ControlRoom

app = typer.Typer(add_completion=False)

@app.command()
def run(
    plan: Path = typer.Option(..., help="Path to plan YAML"),
    env: Path = typer.Option(..., help="Path to environment YAML"),
    headful: bool = typer.Option(True, help="Open visible browser"),
    control_room: bool = typer.Option(False, help="Enable live Control Room"),
    artifacts_dir: Path = typer.Option(Path("artifacts"), help="Artifacts output dir"),
):
    """Run a test plan with optional Control Room."""
    run_id = time.strftime("%Y%m%dT%H%M%S")
    artifacts_dir = artifacts_dir / run_id
    artifacts_dir.mkdir(parents=True, exist_ok=True)

    async def _run():
        cr = None
        if control_room:
            cr = ControlRoom()
            cr.start_in_background()
            print("Control Room ➜ http://127.0.0.1:8788")

        graph = TestGraph(
            artifacts_dir=str(artifacts_dir),
            headful=headful,
            control_room=cr,
            run_id=run_id
        )
        result = await graph.run(str(plan), str(env))
        (artifacts_dir / "run.json").write_text(json.dumps(result, indent=2))

    asyncio.run(_run())

if __name__ == "__main__":
    app()

# NEW imports
import time, asyncio
from orchestrator.control_room import ControlRoom

class Executor:
    def __init__(self, page, context, sink, control_room: ControlRoom|None, run_id: str):
        self.page = page
        self.context = context
        self.sink = sink
        self.cr = control_room
        self.run_id = run_id
        # tap console and network
        self.page.on("console", self._on_console)
        self.page.on("request", self._on_request)
        self.page.on("response", self._on_response)

    def _on_console(self, msg):
        if not self.cr: return
        asyncio.create_task(self.cr.send_log(self.run_id, msg.type, "console", msg.text, time.time()))

    def _on_request(self, req):
        if not self.cr: return
        asyncio.create_task(self.cr.send_log(self.run_id, "info", "network", f"→ {req.method} {req.url}", time.time()))

    def _on_response(self, resp):
        if not self.cr: return
        asyncio.create_task(self.cr.send_log(self.run_id, "info", "network", f"← {resp.status} {resp.url}", time.time()))

    async def _thumb_loop(self):
        # throttled screenshot loop while a step is active
        last = 0
        while self._step_active:
            now = time.time()
            if now - last > 0.6:
                try:
                    png = await self.page.screenshot(full_page=False)
                    await self.cr.send_thumb_png(self.run_id, png, ts=now)
                except Exception:
                    pass
                last = now
            await asyncio.sleep(0.05)

    async def run_step(self, idx: int, step):
        self._step_active = True
        if self.cr:
            await self.cr.send_step(self.run_id, idx, step.title, "executing")
        thumb_task = asyncio.create_task(self._thumb_loop()) if self.cr else None

        try:
            # ... existing step execution logic ...
            # example: click/save/verify
            pass
        except Exception as e:
            if self.cr:
                await self.cr.send_log(self.run_id, "error", "agent", f"Step failed: {e}", time.time())
                await self.cr.send_step(self.run_id, idx, step.title, "failed", str(e))
            raise
        finally:
            self._step_active = False
            if thumb_task:
                thumb_task.cancel()

        if self.cr:
            await self.cr.send_step(self.run_id, idx, step.title, "passed")
Approval pause (e.g., before destructive action)
    async def require_approval(self, reason: str):
        if not self.cr:
            return True
        await self.cr.send_status(self.run_id, "awaiting_approval", reason)
        cmd = await self.cr.wait_for_control(self.run_id, {"approve","reject","stop"})
        if cmd.get("cmd") == "approve":
            await self.cr.send_status(self.run_id, "running", "Approved")
            return True
        elif cmd.get("cmd") == "reject":
            await self.cr.send_status(self.run_id, "paused", "Rejected by user")
            return False
        else:
            await self.cr.send_status(self.run_id, "cancelled", "Stopped by user")
            raise RuntimeError("Run stopped by user")

class TestGraph:
    def __init__(self, artifacts_dir: str, headful: bool, control_room, run_id: str):
        self.artifacts_dir = artifacts_dir
        self.headful = headful
        self.cr = control_room
        self.run_id = run_id

    async def run(self, plan_path: str, env_path: str):
        # setup browser context (video + trace)
        from browser.context import create_context
        browser, context, page = await create_context(headful=self.headful, artifacts_dir=self.artifacts_dir)

        # notify CR
        if self.cr:
            await self.cr.send_status(self.run_id, "starting", "Launching")

        # wire executor with CR
        from evidence.sink import EvidenceSink
        sink = EvidenceSink(self.artifacts_dir)
        from orchestrator.executor import Executor
        exec = Executor(page, context, sink, self.cr, self.run_id)

        # ... load plan/env, iterate steps and exec.run_step(...)
        # finalize
        await context.tracing.stop(path=str(Path(self.artifacts_dir)/"trace.zip"))
        await context.close(); await browser.close()

        if self.cr:
            await self.cr.send_status(self.run_id, "passed", "Done")

        return {"run_id": self.run_id, "status": "passed", "plan_name": "(from plan)", "started_at": time.time()}
 
4) Browser Context: ensure video + trace + headers
browser/context.py
from playwright.async_api import async_playwright
from pathlib import Path

async def create_context(headful: bool, artifacts_dir: str):
    p = await async_playwright().start()
    browser = await p.chromium.launch(headless=not headful)
    context = await browser.new_context(
        record_video_dir=str(Path(artifacts_dir)/"video")
    )
    await context.tracing.start(screenshots=True, snapshots=True, sources=True)
    page = await context.new_page()
    return browser, context, page

import asyncio, json, time
from pathlib import Path
from orchestrator.graph import TestGraph
from orchestrator.control_room import ControlRoom

PLAN = "examples/plan.create_employee.yaml"
ENV = "examples/env.staging.yaml"

async def main():
    run_id = time.strftime("%Y%m%dT%H%M%S")
    artifacts = Path("artifacts")/run_id
    artifacts.mkdir(parents=True, exist_ok=True)

    cr = ControlRoom(); cr.start_in_background()
    graph = TestGraph(str(artifacts), headful=True, control_room=cr, run_id=run_id)
    out = await graph.run(PLAN, ENV)
    (artifacts/"run.json").write_text(json.dumps(out, indent=2))

if __name__ == "__main__":
    asyncio.run(main())
