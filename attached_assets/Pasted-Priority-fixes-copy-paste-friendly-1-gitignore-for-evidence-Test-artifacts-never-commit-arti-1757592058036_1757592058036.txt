3) OpenAI provider (Responses API + JSON mode + backoff)
# providers/openai_provider.py
import json, asyncio, backoff
from openai import OpenAI
client = OpenAI()  # uses OPENAI_API_KEY

@backoff.on_exception(backoff.expo, Exception, max_time=90)
async def plan_fields(chunk: dict, profile: dict, locale: str) -> dict:
    resp = client.responses.create(
        model="gpt-4.1-mini",
        response_format={"type":"json_object"},
        input=[
            {"role":"system","content":"You are a test-data planner. Return strictly valid JSON for values/action."},
            {"role":"user","content=json.dumps({"chunk":chunk,"profile":profile,"locale":locale})}
        ]
    )
    return json.loads(resp.output_text)
(Your README promises “AI Test Generation”; this keeps it robust with modern SDK semantics.)
